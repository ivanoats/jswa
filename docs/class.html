    But that syntax is “a little convoluted, impractical and repetitive” (not classical enough?) for Alex MacCaw, so he makes include and extend methods like so:

    Person.include({
  save:  function(id) {
    console.log( id + ' is being saved');
  },
  destroy: function(id) {
    console.log( id + 'is being destroyed!');
  }
});

var person = new Person();
person.save();

Person.extend({
  extended: function(klass) {
    console.log(klass, ' was extended');
  }
});
